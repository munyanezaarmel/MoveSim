import { NextRequest, NextResponse } from 'next/server';
import { Aptos, AptosConfig } from '@aptos-labs/ts-sdk';
import { MOCK_SIMULATIONS } from '@/lib/mock-data';

const config = new AptosConfig({
  fullnode: "https://testnet.movementnetwork.xyz/v1",
});
const aptos = new Aptos(config);

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { sender, contract, functionName, args, demoMode } = body;

    // DEMO MODE: Return mock data immediately
    if (demoMode) {
      // Determine which mock to return based on function
      const mockKey = functionName.includes('transfer') 
        ? 'token_transfer' 
        : functionName.includes('mint')
        ? 'nft_mint'
        : 'token_transfer';
      
      // Simulate network delay for realism
      await new Promise(resolve => setTimeout(resolve, 500));
      
      return NextResponse.json({
        mode: 'demo',
        ...MOCK_SIMULATIONS[mockKey]
      });
    }

    // LIVE MODE: Real blockchain simulation
    try {
      const payload = {
        function: `${contract}::${functionName}`,
        type_arguments: [],
        arguments: args
      };

      const [simulation] = await aptos.transaction.simulate.simple({
        signerPublicKey: sender,
        transaction: {
          sender,
          data: payload
        }
      });

      // Parse real simulation results
      const result = {
        mode: 'live',
        success: simulation.success,
        gasUsed: parseInt(simulation.gas_used),
        gasCostMOVE: (parseInt(simulation.gas_used) * 0.000001).toFixed(6),
        gasCostUSD: (parseInt(simulation.gas_used) * 0.000001 * 3).toFixed(4),
        executionTime: simulation.gas_used ? parseInt(simulation.gas_used) / 10 : 0,
        stateChanges: parseStateChanges(simulation.changes || []),
        events: simulation.events || [],
        gasBreakdown: {
          computation: Math.floor(parseInt(simulation.gas_used) * 0.4),
          storage: Math.floor(parseInt(simulation.gas_used) * 0.35),
          network: Math.floor(parseInt(simulation.gas_used) * 0.25)
        },
        error: !simulation.success ? {
          type: 'EXECUTION_FAILURE',
          message: simulation.vm_status,
          code: 0
        } : null
      };

      return NextResponse.json(result);

    } catch (blockchainError: any) {
      // If blockchain call fails, return error
      return NextResponse.json({
        mode: 'live',
        success: false,
        error: {
          type: 'BLOCKCHAIN_ERROR',
          message: blockchainError.message || 'Failed to connect to Movement testnet',
          suggestion: 'Try switching to Demo Mode or check your inputs'
        }
      }, { status: 500 });
    }

  } catch (error: any) {
    return NextResponse.json({
      error: error.message || 'Internal server error'
    }, { status: 500 });
  }
}

function parseStateChanges(changes: any[]): any[] {
  // Parse Movement blockchain state changes
  return changes.map(change => ({
    address: change.address,
    resource: change.state_key_hash,
    before: change.data?.before || 'N/A',
    after: change.data?.after || 'N/A',
    change: 'Modified'
  }));
}

import { NextRequest, NextResponse } from 'next/server';

export async function POST(request: NextRequest) {
  const { contract, code } = await request.json();
  
  // Basic security analysis
  const issues = [];
  
  if (code.includes('public entry fun') && !code.includes('assert!')) {
    issues.push({
      severity: 'HIGH',
      type: 'Missing Access Control',
      line: 0,
      message: 'Public function lacks permission checks',
      suggestion: 'Add assert!(signer::address_of(account) == @admin, ERROR_NOT_ADMIN);'
    });
  }
  
  if (code.includes('+ ') && !code.includes('checked_add')) {
    issues.push({
      severity: 'MEDIUM',
      type: 'Potential Integer Overflow',
      line: 0,
      message: 'Arithmetic operation may overflow',
      suggestion: 'Use checked_add() or checked_mul() for safe arithmetic'
    });
  }
  
  return NextResponse.json({ issues });
}
